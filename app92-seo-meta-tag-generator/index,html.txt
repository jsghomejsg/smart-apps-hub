<script>
    function update() {
        const t = document.getElementById('title').value || "Your Site Title";
        const d = document.getElementById('desc').value || "Your site description will appear here...";
        const u = document.getElementById('url').value;
        const k = document.getElementById('keys').value;

        // Update Preview
        document.getElementById('p-title').innerText = t;
        document.getElementById('p-desc').innerText = d;
        document.getElementById('p-url').innerText = u;

        // Generate Code
        let code = `&lt;!-- Primary Meta Tags --&gt;\n`;
        code += `&lt;title&gt;${t}&lt;/title&gt;\n`;
        code += `&lt;meta name="title" content="${t}"&gt;\n`;
        code += `&lt;meta name="description" content="${d}"&gt;\n`;
        if(k) code += `&lt;meta name="keywords" content="${k}"&gt;\n`;
        code += `\n&lt;!-- Open Graph / Facebook --&gt;\n`;
        code += `&lt;meta property="og:type" content="website"&gt;\n`;
        code += `&lt;meta property="og:url" content="${u}"&gt;\n`;
        code += `&lt;meta property="og:title" content="${t}"&gt;\n`;
        code += `&lt;meta property="og:description" content="${d}"&gt;`;

        document.getElementById('code').innerHTML = code;
    }

    function copy() {
        const codeText = document.getElementById('code').innerText;
        navigator.clipboard.writeText(codeText);
        alert("Code copied to clipboard!");
    }

    update();
